"""
Par√¢metros: w - vetor de frequ√™ncias (sugest√£o: usar um vetor com amostras de 0 a 20 rad/s)
            wc - freq. de corte do filtro (em rad/s)
            n - ordem do filtro de Chebyshev
Sa√≠da:      Tn - vetor com os coeficientes calculados do polin√¥mio de Chebyshev (possui o mesmo tamanho que w)
"""

def coeficientes_chebyshev(w, wc, n):
    Tn = np.zeros((w.size,))
    Tn[abs(w) < wc] = np.cos(n*np.arccos(w[abs(w) < wc] / wc))
    Tn[abs(w) >= wc] = np.cosh(n*np.arccosh(w[abs(w) >= wc] / wc))
    return Tn

Implementando uma fun√ß√£o para calcular o m√≥dulo da resposta em frequ√™ncia para o filtro de Chebyshev

def resposta_chebyshev(w, wc, n, e):
    Tn = coeficientes_chebyshev(w, wc, n)

    resposta = 1 / np.sqrt(1 + (e**2) * (Tn**2))

    return resposta

def resposta_fpb_ideal(w, wc):
    resposta = np.copy(w)
    
    for i in range(0, len(w)):
        if np.abs(w[i]) <= wc:
            resposta[i] = 1
        else:
            resposta[i] = 0
    
    return resposta
